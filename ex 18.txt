clear

echo "Digite um número"
read numero

tabuada=$[numero*0]
echo "$numero x 0 = $tabuada"

tabuada=$[numero*1]
echo "$numero x 1 = $tabuada"

tabuada=$[numero*2]
echo "$numero x 2 = $tabuada"

tabuada=$[numero*3]
echo "$numero x 3 = $tabuada"

tabuada=$[numero*4]
echo "$numero x 4 = $tabuada"

tabuada=$[numero*5]
echo "$numero x 5 = $tabuada"

tabuada=$[numero*6]
echo "$numero x 6 = $tabuada"

tabuada=$[numero*7]
echo "$numero x 7 = $tabuada"

tabuada=$[numero*8]
echo "$numero x 8 = $tabuada"

tabuada=$[numero*9]
echo "$numero x 9 = $tabuada"

tabuada=$[numero*10]
echo "$numero x 10 = $tabuada"
